package main

import (
	"flag"
	"fmt"
	"os"
)

func main() {
	for i, arg := range os.Args { // os.Args is []string that holds command line arguments
		fmt.Printf("Argument %d : %s\n", i, arg)
	}

	var name string
	var age int
	var male bool

	flag.StringVar(&name, "name", "John", "Name of the user") // stores the value of -name flag in name variable
	flag.IntVar(&age, "age", 30, "Age of the user")           // stores the value of -age flag in age variable
	flag.BoolVar(&male, "male", true, "Is user a male")       // stores the value of -male flag in male variable

	flag.Parse() // parses the command line flags

	fmt.Println("Name :", name)
	fmt.Println("Age :", age)
	fmt.Println("Male :", male)
}
